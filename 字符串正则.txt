
//去掉左边的空白  
	String.prototype.trimLeft = function(str){  
		return this.replace(/(^\s*)/g, "");  
	}
	
	//去掉右边的空白  
	String.prototype.trimRight = function(){  
		return this.replace(/(\s*$)/g, "");
	}
	
	//去掉两边的空白  
	String.prototype.trim = function(){  
		return this.replace(/(^\s*)|(\s*$)/g, "");  
	}
	//去除所有空格 
	String.prototype.trimAll = function(){  
		return  this.replace(/\s+/g,"");       
	}
